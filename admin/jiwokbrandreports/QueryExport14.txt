SELECT CONCAT(camp_name,'-', cn.camp_value ) AS Brand,CONCAT(a.user_fname,a.user_lname) AS Name,
a.user_email as Mail
FROM `campaign_reports` cr
LEFT JOIN campaign_manage cn ON cr.jiwok_code = cn.camp_id
left join user_master_ext a on a.user_id=cr.user_id
left join countries d on a .user_country=d.countries_id
where cr.status=1 and cr.user_id is not null
and cr.date BETWEEN '%3$s-%2$s-%1$s'
AND '%6$s-%5$s-%4$s'
