SELECT IFNULL( c.brand_name, 'Jiwok' ) AS Brand, IF( `payment_amount` <= 7.9
OR `payment_amount` > 9.9, 'Euro', 'Dollar' ) AS Currency, sum( `payment_amount` ) AS 'payment Amount'
FROM user_master_ext AS a
LEFT JOIN payment p ON a.user_id = p.payment_userid
LEFT JOIN brand_user b ON b.user_id = a.user_id
LEFT JOIN brand_master c ON c.brand_master_id = b.brand_master_id
LEFT JOIN countries d ON a.user_country = d.countries_id
WHERE `payment_status` =1
AND `payment_amount` IS NOT NULL
AND payment_date BETWEEN '%3$s-%2$s-%1$s' and '%6$s-%5$s-%4$s'
